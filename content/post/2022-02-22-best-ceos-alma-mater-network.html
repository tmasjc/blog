library(tidyverse)
library(rvest)
url <- "https://hbr.org/2019/11/the-best-performing-ceos-in-the-world-2019"

tbl_raw <- url |> read_html() |> html_table() 

tbl <- tbl_raw |> 
    pluck(1) |> 
    set_names(c("index", "pic", "name_company", "info")) |> 
    filter(row_number() != n()) |> 
    separate(
        name_company,
        into = c("first", "last", "company"),
        sep = "\\s",
        extra = "merge"
    ) |> 
    separate(
        info,
        into = c(
            "industry",
            "country",
            "start_year",
            "insider",
            "mba",
            "financial_rank",
            "sustainalytics",
            "csrhub"
        ),
        sep = "\\|"
    ) |> 
    mutate(across(
        industry:csrhub,
        ~ str_remove(.x, pattern = "^[\\s]?[A-Za-z]*[\\s]?[A-Za-z]*[:]")
    )) |> 
    mutate(across(industry:csrhub, str_trim)) |> 
    select(-pic)
tbl
