f <- function(i){
    if (i < 10) warning("A warning")
    if (!is.numeric(i)) stop("Not a number!")
    return(i)
}

custom_f <- function(x) {
    tryCatch(
        f(x),
        error = function(e)
            return(list("", e)),
        warning = function(w)
            return(list(f(x), w))
    )
}

custom_f(15) 
## [1] 15
custom_f(9)

## [1] 9
